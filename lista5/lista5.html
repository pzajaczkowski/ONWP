def calculate_triple_integral(X):
    # Granice cakowania
    z_lower = x**2 + y**2
    z_upper = 4
    y_lower = -2
    y_upper = 2
    x_lower = -sqrt(4 - y**2)
    x_upper = sqrt(4 - y**2)

    # Caka potrjna
    integral_result = Integral(
        X, (z, z_lower, z_upper), (x, x_lower, x_upper), (y, y_lower, y_upper)
    )
    display(integral_result)
    intergral_result_val = integral_result.doit()

    return intergral_result_val
